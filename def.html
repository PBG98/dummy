import requests from bs4 import BeautifulSoup for i in range(1, 10, 1): response = requests.get(f"https://hiphopplaya.com/g2/bbs/board.php?bo_table=hiphoptalk&page={i}", verify=False) html = response.text soup = BeautifulSoup(html, 'html.parser') links = soup.select(".list_subject_a") print(f"{i}") for link in links: title = link.text.strip() print(title) url = link.attrs['href'] print(url)
